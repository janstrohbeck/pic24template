35 void lcd_cmd (uint8_t u8cmd)
36 {
37  RS = 0; // Command mode
38  CSB = 0; // Select display
39  LCD_BUF; // Read Buffer -> Clear RBF bit
40  LCD_BUF = u8cmd; // Write command bits
41  LCD_BUF; // Read Buffer -> Clear RBF bit
42 
43  while (LCD_STATbits.SPITBF); // wait until all bits are sent
44  uint16_t i;
45  for (i = 0; i < 150; i++); // Create a small delay
46  CSB = 1; // Unselect display
47 }

92 void lcd_display_long_string (c8_t *pac8string, uint16_t u16delay)
93 {
94  // Used later to remember the passed string.
95  static c8_t *pc8char = NULL;
96  // Array to hold our single lines
97  static c8_t *ac8buf[30];
98  // Line count
99  static uint8_t u8count = 0;
100  // Array index
101  uint8_t u8index = 0;
102  // If the string hasn't already been parsed
103  if (pc8char == NULL || pc8char != pac8string)
104  {
105  pc8char = pac8string;
106 
107  // Split string by using linebreaks as delimiters
108  pc8char = strtok (pac8string, "\n");
109 
110  // While there are still linebreaks in the string
111  while (pc8char != NULL)
112  {
113  // Store the line in the array
114  ac8buf[u8index++] = pc8char;
115  // Get next line in string
116  pc8char = strtok (NULL, "\n");
117  // Increment Line count
118  u8count++;
119  }
120  // Remember this string
121  pc8char = pac8string;
122  }
123 
124  u8index = 0;
125  // to handle two \n's in sequence: Insert a newline
126  bool b8skipline = FALSE;
127  // Iterate over the lines
128  while (u8index < u8count)
129  {
130  lcd_clear ();
131  // If the line does not have to be clear because of a second newline
132  if (!b8skipline)
133  // Display the line
134  lcd_display_string (ac8buf[u8index]);
135  // Go to the second line
136  lcd_set_cursor (1, 0);
137  // If the previous line didn't have to be clear
138  if (!b8skipline)
139  // Look if the line has a second \n
140  if (*(ac8buf[u8index+1]-1) == '\n')
141  // Then skip this line and next time, put this line in the second line
142  b8skipline = TRUE;
143  else
144  // Otherwise display the line
145  lcd_display_string (ac8buf[++u8index]);
146  else
147  {
148  // Otherwise display the line in the second line instead of the first line.
149  lcd_display_string (ac8buf[++u8index]);
150  // Finally done.
151  b8skipline = FALSE;
152  }
153  delayMs (u16delay);
154  }
155 }
