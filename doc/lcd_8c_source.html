33 void lcd_cmd (uint8_t u8cmd)
34 {
35  CSB = 0; // Beginning of data transfer
36  RS = 0; // Command mode
37  LCD_BUF &= 0xFF00; // prepare RD0 - RD7
38  LCD_BUF |= u8cmd; // Write command bits
39 
40  while (LCD_STATbits.SPITBF); // wait until all bits are sent
41  CSB = 1; // End of data transfer
42  delayMs (100);
43 }

45 void lcd_display_char (c8_t c8character)
46 {
47  CSB = 0; // Beginning of data transfer
48  RS = 1; // Data mode
49  LCD_BUF &= 0xFF00; // prepare RD0 - RD7
50  LCD_BUF |= c8character; // Write data bits
51 
52  // Test -> Oscilloscope
53  if (LCD_STATbits.SPITBF)
54  LATB |= (1<<15);
55 
56  while (LCD_STATbits.SPITBF); // wait until all bits are sent
57 
58  LATB &= ~(1<<15);
59 
60  RS = 0; // Quit data mode
61  CSB = 1; // End of data transfer
62  delayMs (100);
63 }

65 void lcd_init (void)
66 {
67  RPOR2 = 0x708; // Use RP4 as SCK1 and RP5 as SDO1
68 
69  TRISB &= ~(1<<4); // SCK -> Output
70  TRISB &= ~(1<<5); // SDO -> Output
71  TRISB &= ~(1<<7); // SS -> Output
72  TRISB |= (1<<6); // SDI -> Input
73  TRISA &= ~(1<<3); // RS -> Output
74  TRISA &= ~(1<<4); // CSB -> Output
75 
76  CSB = 1;
77 
78  LATB |= (1<<7); // Make Slave listen (SS)
