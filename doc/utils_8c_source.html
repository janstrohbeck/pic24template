38 void LED ( volatile uint16_t *u16ledtris, volatile uint16_t *u16ledlat, uint8_t u8led, uint8_t u8action )
39 {
40  switch ( u8action )
41  {
42  case ON:
43  *u16ledlat |= (1 << u8led);
44  break;
45  case OFF:
46  *u16ledlat &= ~(1 << u8led);
47  break;
48  case TOGGLE:

51  case ENABLED:
52  *u16ledtris &= ~(1 << u8led);
53  break;
54  case DISABLED:
55  default:
56  *u16ledtris |= (1 << u8led);
